### 3. 결제 시스템 아키텍처

```typescript
// 빌링키 발급
const billingKey = await tossPayments.issueBillingKey(cardInfo);

// 매월 자동 청구 (Cron)
@Cron('0 0 1 * *') // 매월 1일
async processMonthlyBilling() {
  const subscriptions = await getActiveSubscriptions();
  for (const sub of subscriptions) {
    await tossPayments.charge(sub.billingKey, sub.plan.totalPrice);
  }
}
```
